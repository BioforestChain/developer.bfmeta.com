```typescript

```
- 返回参数：
```typescript

    /**成功 */
    success: true;
    /**最小手续费 */
    minFee: number;
    /**完整的转移事件 */
    result: BFMetaCore.TransactionJSON<BFMetaCore.TransferAssetAssetJSON>;

    /**失败 */

    /**错误信息 */
